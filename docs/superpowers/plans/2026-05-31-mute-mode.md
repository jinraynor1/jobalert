# Mute Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Agregar botones "1h / 2h / 8h" a la notificación persistente para silenciar el overlay durante una avería, guardando las alertas en historial pero sin mostrar el popup.

**Architecture:** `NotificationHelper` centraliza la construcción de la notificación con botones de acción. `SilenceReceiver` persiste `muteUntil` en SharedPreferences y programa `AlarmManager` para la expiración automática. El servicio lee `isMuted` antes de llamar al overlay. `NotificationHelper` y `SilenceReceiver` se crean juntos en un solo task porque tienen dependencia circular de compilación (`NotificationHelper` crea `Intent(context, SilenceReceiver::class.java)`).

**Tech Stack:** Android BroadcastReceiver, AlarmManager, PendingIntent, NotificationCompat actions, SharedPreferences, Room Flow.

## Task 2: NotificationHelper.kt + SilenceReceiver.kt (ambos juntos — dependencia circular)

**Files:**
- Create: `app/src/main/java/com/jobalert/service/NotificationHelper.kt`
- Create: `app/src/main/java/com/jobalert/service/SilenceReceiver.kt`

Estos dos archivos deben crearse en el mismo task porque `NotificationHelper` referencia `SilenceReceiver::class.java` para los `PendingIntent`, y `SilenceReceiver` llama a `buildServiceNotification()` de `NotificationHelper`. Si se crean por separado el build falla en el paso intermedio.

- [ ] **Paso 1: Crear NotificationHelper.kt**

  Crear `app/src/main/java/com/jobalert/service/NotificationHelper.kt` con el contenido exacto:

  ```kotlin
  package com.jobalert.service

  import android.app.Notification
  import android.app.PendingIntent
  import android.content.Context
  import android.content.Intent
  import androidx.core.app.NotificationCompat
  import com.jobalert.JobAlertApp
  import com.jobalert.R
  import com.jobalert.ui.MainActivity
  import java.text.SimpleDateFormat
  import java.util.Date
  import java.util.Locale

  internal const val RC_SILENCE_1H = 10
  internal const val RC_SILENCE_2H = 11
  internal const val RC_SILENCE_8H = 12
  internal const val RC_UNSILENCE  = 13
  internal const val RC_EXPIRED    = 14

  fun buildServiceNotification(
      context: Context,
      unacknowledgedCount: Int,
      isMuted: Boolean,
      muteUntil: Long
  ): Notification {
      val openAppIntent = Intent(context, MainActivity::class.java).apply {
          flags = Intent.FLAG_ACTIVITY_SINGLE_TOP
      }
      val openAppPendingIntent = PendingIntent.getActivity(
          context, 0, openAppIntent, PendingIntent.FLAG_IMMUTABLE
      )

      val builder = NotificationCompat.Builder(context, JobAlertApp.CHANNEL_ID)
          .setSmallIcon(R.drawable.ic_notification)
          .setContentTitle("JobAlert activo")
          .setContentIntent(openAppPendingIntent)
          .setOngoing(true)
          .setPriority(NotificationCompat.PRIORITY_LOW)

      if (isMuted) {
          val timeStr = SimpleDateFormat("HH:mm", Locale.getDefault()).format(Date(muteUntil))
          builder.setContentText("Silenciado hasta $timeStr")
          builder.addAction(
              0, "Reactivar",
              silencePendingIntent(context, SilenceReceiver.ACTION_UNSILENCE, RC_UNSILENCE, 0L)
          )
      } else {
          val text = if (unacknowledgedCount == 0) "Escuchando notificaciones de Gmail"
                     else "$unacknowledgedCount alerta(s) sin atender"
          builder.setContentText(text)
          builder.addAction(
              0, "1h",
              silencePendingIntent(context, SilenceReceiver.ACTION_SILENCE, RC_SILENCE_1H, 60 * 60 * 1000L)
          )
          builder.addAction(
              0, "2h",
              silencePendingIntent(context, SilenceReceiver.ACTION_SILENCE, RC_SILENCE_2H, 2 * 60 * 60 * 1000L)
          )
          builder.addAction(
              0, "8h",
              silencePendingIntent(context, SilenceReceiver.ACTION_SILENCE, RC_SILENCE_8H, 8 * 60 * 60 * 1000L)
          )
      }

      return builder.build()
  }

  private fun silencePendingIntent(
      context: Context,
      action: String,
      requestCode: Int,
      durationMs: Long
  ): PendingIntent {
      val intent = Intent(context, SilenceReceiver::class.java).apply {
          this.action = action
          putExtra(SilenceReceiver.EXTRA_DURATION_MS, durationMs)
      }
      return PendingIntent.getBroadcast(
          context, requestCode, intent,
          PendingIntent.FLAG_IMMUTABLE or PendingIntent.FLAG_UPDATE_CURRENT
      )
  }
  ```

- [ ] **Paso 2: Crear SilenceReceiver.kt**

  Crear `app/src/main/java/com/jobalert/service/SilenceReceiver.kt` con el contenido exacto:

  ```kotlin
  package com.jobalert.service

  import android.app.AlarmManager
  import android.app.NotificationManager
  import android.app.PendingIntent
  import android.content.BroadcastReceiver
  import android.content.Context
  import android.content.Intent
  import com.jobalert.JobAlertApp
  import kotlinx.coroutines.flow.first
  import kotlinx.coroutines.runBlocking

  class SilenceReceiver : BroadcastReceiver() {

      companion object {
          const val ACTION_SILENCE = "com.jobalert.SILENCE"
          const val ACTION_UNSILENCE = "com.jobalert.UNSILENCE"
          const val ACTION_MUTE_EXPIRED = "com.jobalert.MUTE_EXPIRED"
          const val EXTRA_DURATION_MS = "duration_ms"
      }

      override fun onReceive(context: Context, intent: Intent) {
          val app = context.applicationContext as JobAlertApp
          val settings = app.settingsRepository
          val notificationManager = context.getSystemService(NotificationManager::class.java)

          when (intent.action) {
              ACTION_SILENCE -> {
                  val durationMs = intent.getLongExtra(EXTRA_DURATION_MS, 0L)
                  if (durationMs <= 0L) return
                  val muteUntil = System.currentTimeMillis() + durationMs
                  settings.muteUntil = muteUntil
                  scheduleExpiry(context, muteUntil)
                  notificationManager.notify(
                      JobAlertApp.SERVICE_NOTIFICATION_ID,
                      buildServiceNotification(
                          context = context,
                          unacknowledgedCount = 0,
                          isMuted = true,
                          muteUntil = muteUntil
                      )
                  )
              }
              ACTION_UNSILENCE -> {
                  settings.muteUntil = 0L
                  cancelExpiry(context)
                  val count = runBlocking { app.alertRepository.unacknowledgedCount.first() }
                  notificationManager.notify(
                      JobAlertApp.SERVICE_NOTIFICATION_ID,
                      buildServiceNotification(
                          context = context,
                          unacknowledgedCount = count,
                          isMuted = false,
                          muteUntil = 0L
                      )
                  )
              }
              ACTION_MUTE_EXPIRED -> {
                  settings.muteUntil = 0L
                  val count = runBlocking { app.alertRepository.unacknowledgedCount.first() }
                  notificationManager.notify(
                      JobAlertApp.SERVICE_NOTIFICATION_ID,
                      buildServiceNotification(
                          context = context,
                          unacknowledgedCount = count,
                          isMuted = false,
                          muteUntil = 0L
                      )
                  )
              }
          }
      }

      private fun scheduleExpiry(context: Context, muteUntil: Long) {
          context.getSystemService(AlarmManager::class.java)
              .setAndAllowWhileIdle(AlarmManager.RTC_WAKEUP, muteUntil, expiryPendingIntent(context))
      }

      private fun cancelExpiry(context: Context) {
          context.getSystemService(AlarmManager::class.java)
              .cancel(expiryPendingIntent(context))
      }

      private fun expiryPendingIntent(context: Context): PendingIntent {
          val intent = Intent(context, SilenceReceiver::class.java).apply {
              action = ACTION_MUTE_EXPIRED
          }
          return PendingIntent.getBroadcast(
              context, RC_EXPIRED, intent,
              PendingIntent.FLAG_IMMUTABLE or PendingIntent.FLAG_UPDATE_CURRENT
          )
      }
  }
  ```

- [ ] **Paso 1: Reemplazar el contenido completo del archivo**

  ```kotlin
  package com.jobalert.service

  import android.app.Notification
  import android.app.NotificationManager
  import android.content.pm.ServiceInfo
  import android.os.Build
  import android.service.notification.NotificationListenerService
  import android.service.notification.StatusBarNotification
  import android.util.Log
  import com.jobalert.JobAlertApp
  import com.jobalert.data.model.AlertEntity
  import com.jobalert.domain.NotificationData
  import com.jobalert.rules.RuleEngine
  import kotlinx.coroutines.CoroutineScope
  import kotlinx.coroutines.Dispatchers
  import kotlinx.coroutines.SupervisorJob
  import kotlinx.coroutines.cancel
  import kotlinx.coroutines.launch

  private const val TAG = "JobAlert"

  class AlertNotificationListenerService : NotificationListenerService() {

      private val serviceScope = CoroutineScope(Dispatchers.Main + SupervisorJob())
      private val seenKeys = mutableSetOf<String>()
      private val notificationManager by lazy {
          getSystemService(NOTIFICATION_SERVICE) as NotificationManager
      }

      private val app get() = application as JobAlertApp

      override fun onListenerConnected() {
          super.onListenerConnected()
          Log.i(TAG, "SERVICE CONNECTED — escuchando notificaciones")
          startForegroundCompat(
              buildServiceNotification(
                  context = this,
                  unacknowledgedCount = 0,
                  isMuted = app.settingsRepository.isMuted,
                  muteUntil = app.settingsRepository.muteUntil
              )
          )
          serviceScope.launch {
              app.alertRepository.unacknowledgedCount.collect { count ->
                  notificationManager.notify(
                      JobAlertApp.SERVICE_NOTIFICATION_ID,
                      buildServiceNotification(
                          context = this@AlertNotificationListenerService,
                          unacknowledgedCount = count,
                          isMuted = app.settingsRepository.isMuted,
                          muteUntil = app.settingsRepository.muteUntil
                      )
                  )
              }
          }
      }

      override fun onListenerDisconnected() {
          super.onListenerDisconnected()
          Log.w(TAG, "SERVICE DISCONNECTED — se perdió el enlace con el sistema")
      }

      private fun startForegroundCompat(notification: Notification) {
          if (Build.VERSION.SDK_INT >= 34) {
              startForeground(
                  JobAlertApp.SERVICE_NOTIFICATION_ID,
                  notification,
                  ServiceInfo.FOREGROUND_SERVICE_TYPE_SPECIAL_USE
              )
          } else {
              startForeground(JobAlertApp.SERVICE_NOTIFICATION_ID, notification)
          }
      }

      override fun onNotificationPosted(sbn: StatusBarNotification) {
          Log.d(TAG, "[1] onNotificationPosted: pkg=${sbn.packageName} key=${sbn.key}")

          if (sbn.packageName != "com.google.android.gm") {
              Log.d(TAG, "[1] SKIP: no es Gmail (pkg=${sbn.packageName})")
              return
          }

          if (sbn.notification.flags and Notification.FLAG_GROUP_SUMMARY != 0) {
              Log.d(TAG, "[2] SKIP: es notificación de resumen de grupo (FLAG_GROUP_SUMMARY)")
              return
          }

          if (sbn.key in seenKeys) {
              Log.d(TAG, "[3] SKIP: clave ya procesada key=${sbn.key}")
              return
          }
          seenKeys.add(sbn.key)

          val extras = sbn.notification.extras
          val sender = extras.getString(Notification.EXTRA_TITLE) ?: ""
          val rawSubject = extras.getString(Notification.EXTRA_TEXT) ?: ""
          val bigText = extras.getCharSequence(Notification.EXTRA_BIG_TEXT)?.toString()
          val textLines = extras.getCharSequenceArray(Notification.EXTRA_TEXT_LINES)
              ?.joinToString(" ") { it.toString() }

          // Gmail a veces deja EXTRA_TEXT vacío y pone "asunto\ncuerpo" en EXTRA_BIG_TEXT
          val subject = if (rawSubject.isNotBlank()) {
              rawSubject
          } else {
              bigText?.lines()?.firstOrNull()?.takeIf { it.isNotBlank() } ?: ""
          }
          val snippet = bigText ?: textLines ?: subject

          Log.i(TAG, "[4] Gmail recibido — sender='$sender' | subject='$subject' | snippet='${snippet.take(80)}'")

          val data = NotificationData(sender = sender, subject = subject, snippet = snippet)

          serviceScope.launch {
              val rules = app.ruleRepository.getAllRulesOnce()
              Log.d(TAG, "[5] Reglas cargadas: ${rules.size} regla(s) — $rules")
              val matched = RuleEngine.match(data, rules)
              Log.i(TAG, "[5] RuleEngine.match=$matched")
              if (matched) {
                  Log.i(TAG, "[6] ALERTA DISPARADA — guardando en historial")
                  app.alertRepository.insert(
                      AlertEntity(
                          timestamp = System.currentTimeMillis(),
                          sender = sender,
                          subject = subject,
                          snippet = snippet
                      )
                  )
                  if (!app.settingsRepository.isMuted) {
                      Log.i(TAG, "[6] mostrando overlay")
                      app.overlayManager.show(data)
                  } else {
                      Log.i(TAG, "[6] SILENCIADO — overlay omitido")
                  }
              } else {
                  Log.i(TAG, "[5] Ninguna regla coincidió — sin alerta")
              }
          }
      }

      override fun onNotificationRemoved(sbn: StatusBarNotification) {
          seenKeys.remove(sbn.key)
      }

      override fun onDestroy() {
          super.onDestroy()
          serviceScope.cancel()
          seenKeys.clear()
      }
  }
  ```
